function parseLogs(text) { return text.trim().split('\n').map(line => { const [preRECV, logData] = line.split('RECV'); const type = preRECV.slice(preRECV.lastIndexOf('/') + 1).trim(); return [type, logData.trim()]; }); } function convertTimestamp(timestamp) { const date = new Date(timestamp); const year = date.getFullYear(); const month = String(date.getMonth() + 1).padStart(2, '0'); const day = String(date.getDate()).padStart(2, '0'); const hours = String(date.getHours()).padStart(2, '0'); const minutes = String(date.getMinutes()).padStart(2, '0'); const seconds = String(date.getSeconds()).padStart(2, '0'); return `${year}-${month}-${day} ${hours}:${minutes}:${seconds}`; } const pdtTimestamp = "2023-03-20T10:06:00.315-07:00"; // PDT timezone offset is -07:00 const formattedDate = convertTimestamp(pdtTimestamp); console.log(formattedDate);
